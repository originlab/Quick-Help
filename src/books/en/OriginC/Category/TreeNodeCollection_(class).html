       tree.AddNumericNode(2);
       tree.AddNumericNode(3, "I333");
       tree.AddNumericNode(4);
       
       TreeNodeCollection tnc(tree, "int");
       foreach(TreeNode tn in tnc)
       {
           string str = tn.tagName;
           out_str(str);
       }
    }    
}
