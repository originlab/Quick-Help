Description

Get numeric display string from column format and display settings

Syntax

string GetNumericDisplayString( double rValue )

Parameters

rValue
[input] double value to convert

Return

the numeric display string.

Examples

EX1 

void    DataObject_GetNumericDisplayString_Ex1()
{
    Worksheet wks;
    wks.Create("origin", CREATE_VISIBLE);
 
    Column col(wks, 1);
    col.SetFormat(OKCOLTYPE_TIME);
    
    string strDisplay = col.GetNumericDisplayString(0.1);
    printf("GetNumericDisplayString: %s", strDisplay);
}
