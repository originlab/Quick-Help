Name

GETN_ID_BRANCH

Declaration

#define GETN_ID_BRANCH(_ID) _tmpNode.DataID = _ID;

Remark

Set the branch DataID, this DataID will use to load and save theme settings.

Parameters

_ID
DataID will set to branch node

Return

Examples

EX1 

#include <GetNbox.h>
#define BRANCH_DATAID  12
void GETN_ID_BRANCH_ex1()
{
    GETN_TREE(tr)

    GETN_BEGIN_BRANCH(WhichDay, "Test Branch with Combo") GETN_ID_BRANCH( BRANCH_DATAID)
        GETN_LIST(Smooth, "Smoothing Method", 0, "Adj Ave|Savisky-Golay|FFT Filter")
        GETN_COMBO(npts, "Smooth Pts",1, "|1|2|3|4|5|6")
    GETN_END_BRANCH(WhichDay)    //Match GETN_BEGIN_BRANCH

    if(GetNBox(tr, NULL, NULL, NULL, NULL))
    {
        out_int("Test Combo = ", tr.WhichDay.Use);
    }
}
