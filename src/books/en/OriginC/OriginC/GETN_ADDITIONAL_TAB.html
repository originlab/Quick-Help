Name

GETN_ADDITIONAL_TAB

Declaration

#define GETN_ADDITIONAL_TAB(_NTAB) _strTemp = _NTAB;_tmpSubNode.SetAttribute(STR_ATTRIB_ADDITIONAL_TAB, _strTemp);

Remark

The macro allow user to justify controls on dialog

Parameters

_NTAB
[input] Number of tabs to pad ahead this control

Return

Examples

EX1 

#include <GetNbox.h>
void GETN_ADDITIONAL_TAB_ex1()
{
    GETN_BOX(trRoot)
    GETN_CHECK(One, "One", false) GETN_ADDITIONAL_TAB(1)
    GETN_CHECK(Two, "Two", false) GETN_ADDITIONAL_TAB(3)
    GETN_CHECK(Three, "Three", false) GETN_ADDITIONAL_TAB(2)
    GETN_CHECK(Four, "Four", false) GETN_ADDITIONAL_TAB(4)
    if ( GetNBox(trRoot) )
        out_tree(trRoot);
    return;
}
