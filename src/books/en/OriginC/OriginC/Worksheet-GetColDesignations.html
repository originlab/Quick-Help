Description

Get the column designations for a worksheet in string form


Gets column designations for entire worksheet

Syntax

string GetColDesignations( )


void GetColDesignations( string & strDesignations )

Parameters

strDesignations
[output] Receives the column designations

Return

column designations for entire worksheet in form of string


void

Examples

EX1 

// Display the column desgnations of the first worksheet in a project
void Worksheet_GetColDesignations_Ex1()
{
    WorksheetPage wp = Project.WorksheetPages(0);
    if(!wp)
        return;
    
    Worksheet wks(wp.GetName());
    if(wks.IsValid())
    {
        string strDesignations = wks.GetColDesignations();
        printf("%s\n", strDesignations);
    }
    else
        printf("No worksheet found\n");
}


EX2 

// Worksheet must exist in project
void Worksheet_GetColDesignations_Ex2()
{
    WorksheetPage wp = Project.WorksheetPages(0);
    if(!wp)
        return;
    
    Worksheet wks(wp.GetName());
    string strDesignations;
    wks.GetColDesignations(strDesignations);
    printf("%s\n", strDesignations);
}

Remark

Gets column designations for entire worksheet in form of string, no separator

Possible values for designations:

'X' - X column

'Y' - Y column

'Z' - Z column

'E' - Y error column

'L' - label column

'M' - X error column

'N' - ignore column
