EX1 

void    DataObjectBase_i2_ex1()
{
    Worksheet wks;
    wks.Create("origin");
    Dataset ds1(wks,0);
    Dataset ds2(wks,1);
    ds1.Data(1,20,1);
    ds2.Normal(20);
    GraphPage gp;
    gp.Create("origin");
    GraphLayer gl(gp.GetName(), 0);
    gl.AddPlot(wks);
    
    // Get the first data plot in the layer:
    DataPlot        dp = gl.DataPlots(0);
    int                i2 = dp.i2;

    printf("Data plot range: i2 = %d\n", i2);
}

Description

The upper data range index for the object.
