Description

Get the column internal data type.

Syntax

UINT GetInternalData( )

Parameters

Return

Data type enum FSITYPE {

FSI_DOUBLE,
FSI_REAL,
FSI_SHORT,
FSI_LONG,
FSI_CHAR,
FSI_TEXT,
FSI_MIXED,
FSI_BYTE,
FSI_USHORT,
FSI_ULONG,
FSI_COMPLEX,

};

Examples

EX1 

// Assume worksheet is the active window
void Column_GetInternalData_Ex1()
{
    Worksheet wks(Project.ActiveLayer());
    if(!wks)
        return;
    int iInternalData = wks.Columns(0).GetInternalData();
    printf("Internal type of column 1 of worksheet %s is %u\n", wks.GetName(), iInternalData);
}
