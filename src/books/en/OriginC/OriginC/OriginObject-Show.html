Remark

Read/Write the Show/Hide property of any Origin object

Examples

EX1 

// Create a worksheet with six columns but hide the 'odd' columns
void OriginObject_Show_Ex1()
{
    Worksheet wks;
    wks.Create("origin", CREATE_VISIBLE);
    if (wks)
    {
        for( int ii = 1 ; ii <= 4 ; ii++ )
            wks.AddCol();
        for( ii = 0 ; ii < wks.GetNumCols() ; ii += 2 )
            wks.Columns(ii).Show = 0;    //hide 'odd' columns
    }
}

EX2 

// to hide the first worksheet
void OriginObject_Show_Ex2()
{
    int nSheetIndex = 0;
    WorksheetPage wksPage = Project.Pages();
    Worksheet wks = wksPage.Layers(nSheetIndex);
    if( wks )
    {
        wks.Show = true;   
    }   
}
