Description

It retrieves the data (display) range for the object.

Syntax

BOOL GetRange( int & i1, int & i2 )

Parameters

i1
[output] lower boundary of the range passed by reference to receive the result
i2
[output] upper boundary of the range passed by reference to receive the result

Return

TRUE for success, FALSE for failure.

Examples

EX1 

void    DataObjectBase_GetRange_ex1()
{
    Worksheet wks;
    wks.Create("origin");
    Dataset ds1(wks,0);
    Dataset ds2(wks,1);
    ds1.Data(1,20,1);
    ds2.Normal(20);

    GraphPage gp;
    gp.Create("origin");

    GraphLayer gl(gp.GetName(), 0);
    gl.AddPlot(wks);
    
    // Get the first data plot in the layer:
    DataPlot        dp = gl.DataPlots(0);
    int                i1, i2;

    dp.GetRange(i1, i2);
    
    printf("Data plot range: i1 = %d   i2 = %d\n", i1, i2);
}
