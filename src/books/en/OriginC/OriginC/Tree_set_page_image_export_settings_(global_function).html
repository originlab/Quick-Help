Description

Set the image export settings into a page.

Syntax

bool tree_set_page_image_export_settings( TreeNode & trSettings, Page & pg, LPCSTR lpcszFormat )

Parameters

trSettings
[input]tree node containing the settings
pg
[modify]page to set the settings into
lpcszFormat
[input]pointer to the image format

Return

true for success, false for error

Examples

EX1 

//First should update image export settings, see the description in tree_read_image_export_settings's example.
void tree_set_page_image_export_settings_ex1()
{
	Tree trSettings;
	bool bRet = tree_read_image_export_settings(trSettings, "bmp");
	Page pg = Project.Pages(); //active page
	if ( bRet )
	{
		bRet = tree_set_page_image_export_settings(trSettings, pg, "bmp");
		if ( bRet )
			out_str("Set page image export settings successfully.");
		else
			out_str("Fail to set image export settings into page.");
	}
	return;
}
