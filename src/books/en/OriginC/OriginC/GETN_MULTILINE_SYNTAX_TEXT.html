Name

GETN_MULTILINE_SYNTAX_TEXT

Declaration

#define	GETN_MULTILINE_SYNTAX_TEXT(_NODE_NAME, _NODE_LABEL, _DEFAULT_VAL)	_tmpSubNode = _tmpNode.AddTextNode(_DEFAULT_VAL, #_NODE_NAME, TRGP_SYNTAX_EDIT);TREE_ADD_LABEL(_NODE_LABEL);

Remark

multiline syntax edit box

Parameters

_NODE_NAME
[input] the name of the new node to be added to the tree.
_NODE_LABEL
[input] the string value to show the edit box name in the dialogbox.
_DEFAULT_VAL
[input] string contents as the default value to show in the edit box.

Return

Examples

EX1 

void GETN_MULTILINE_SYNTAX_TEXT_Ex1()
{
    GETN_TREE(tr)
    GETN_MULTILINE_SYNTAX_TEXT(BeforeLoop, _L("Script Before Select Data Loop"), "1+(a*(b-(c-3)+d))")
        GETN_MULTILINE_EDIT_DISPLAY_ROW_HEIGHT_RANGE("5-10")
        GETN_EDIT_DISPLAY_WIDTH_RANGE( "40-40" )
    if( GetNBox(tr ))
        out_tree( tr );
}
