Description

The Active CurveBase in the active layer of the active page(window)

Syntax

curvebase & ActiveCurveBase( )

Parameters

Return

the active CurveBase object

Examples

EX1 

void    run_Project_ActiveCurveBase()
{
    using cuvB = Project.ActiveCurveBase();
    if(cuvB.IsValid())
    {
        Curve cc(cuvB);
        Dataset dx;
        printf("Active Dataset is %s",cc.GetName());
        if(cc.AttachX(dx, FALSE))
            printf(", its corresponding X Dataset is %s",dx.GetName());
        
        printf(".\n");
    }
    else
        out_str("There is no active dataset");
}
