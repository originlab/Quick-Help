Description

Set marker lines in layer.

Syntax

BOOL SetMarkerLines( vector & vPos, BOOL bShow = TRUE, BOOL bVertical = TRUE )

Parameters

vPos
[input] marker lines position
bShow
[input] show or hide, default:TRUE
bVertical
[input] vertical or horizontal, default:TRUE

Return

TRUE if success, otherwise FALSE

Examples

EX1 

//assume there is a row numbers symbol plot
void GraphLayer_SetMarkerLines_ex1()
{
    GraphLayer gl = Project.ActiveLayer();
    if( !gl.IsValid() )
        return;
    
    vector     vPos;
    vPos.Add(1);
    vPos.Add(4);
    BOOL bSuccess = gl.SetMarkerLines(vPos, TRUE, TRUE);
    printf("MoveMarkerLine success: %d\n", bSuccess);
}
