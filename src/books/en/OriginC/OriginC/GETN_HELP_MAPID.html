Name

GETN_HELP_MAPID

Declaration

#define GETN_HELP_MAPID(_STR)		_tmpNode.SetAttribute(STR_DOCUMENT_MAPID_ATTRIB, _STR);

Remark

Specify the page by ID in the help file to be opened by clicking on the question button on the upper right of the dialog.

Parameters

_STR
a string containing the ID number of a specified page in CHM help file.

Return

Examples

EX1 

#include <GetNBox.h>
void GetNBox_ex2()
{
    double x0, x1;
    GETN_BOX(trGetN)
    GETN_HELP("F:\\Help\\X-Function.chm")
    GETN_HELP_MAPID(37023)
    GETN_NUM(xFrom, "X From", 1.3)
    GETN_NUM(xStep, "X Step", -0.5)
    if(GetNBox(trGetN, "Row# as X", "Please specify initial X value and increment"))
    {
        x0 = trGetN.xFrom.dVal;
        x1 = trGetN.xStep.dVal;
        printf("X from %f with increment %f\n", x0, x1);
    }
}
