Examples

VBA

Dim myobj As Origin.Application

Dim mypage As Origin.Page

Dim myWks As Origin.Worksheet

Dim myCol As Origin.Column

Dim strStroage As String

'Open a new instance of the Origin application
Set myobj = GetObject("", "Origin.ApplicationSI")

'
myobj.Execute ("doc -mc 1")

myobj.NewProject

Set mypage = myobj.Pages(0)

'base on 0
Set myWks = mypage.Layers(0)

'start from 0
Set myCol = myWks.Columns(1)

strStorage = "Data"

'set BinaryStorageData

Set myRng = Range("C20:C30")

'get GetBinaryStroageData
myRng.Value = myCol.GetBinaryStorageData(strStroage, STGDATAFMT_BLOB)
