Examples

VB

    Sub Add()
        Dim app As Origin.IOApplication
        Dim dr As Origin.DataRange
        Dim wks As Origin.Worksheet

        app = New Origin.ApplicationSI
        wks = app.FindWorksheet("") ' get the active wks
        If wks Is Nothing Then
            MsgBox("Can not get an active worksheet")
            Exit Sub
        End If
        dr = wks.NewDataRange()
        dr.Add("X", wks, 0, 0, -1, 0) 'add first column of wks as X
        dr.Add("Y", wks, 0, 1, -1, 1) 'add sceond column as Y
    End Sub
