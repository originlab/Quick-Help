Remark

Examples

Python

import OriginExt as O
app = O.Application(); app.Visible = app.MAINWND_SHOW
app.CreatePage(app.OPT_WORKSHEET)
for pg in app.WorksheetPages:
	print(pg.Name)


import win32com.client # COM module

# Connect to Origin
origin = win32com.client.Dispatch("Origin.ApplicationSI")

# Get the number of workbooks in the current project
workbookCount = origin.WorksheetPages.Count
print "Current project contains", workbookCount, "workbooks"

# Get each workbook and output it's name
for i in range(0, workbookCount):
    workbook = origin.WorksheetPages(i)
    print i,"-",workbook.Name
