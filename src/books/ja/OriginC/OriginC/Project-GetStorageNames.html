Description

Get names of storage class objects

Syntax

BOOL GetStorageNames( vector<string> & vsNames )

Parameters

vsNames
[output]a string vector to get storagenames

Return

Examples

//Output the storage name of the active page.
void OriginObject_GetStorageNames_Ex1()
{
        bool bRet;
        vector<string> vsNames;
        Page pg = Project.Pages(-1);
        bRet = pg.GetStorageNames(vsNames);
        if(bRet)
                out_str("Success!");
        for(int ii=0; ii<vsNames.GetSize(); ii++)
                out_str(vsNames[ii]);
}
