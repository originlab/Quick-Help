void Window_FontName_ex1()
{

    pg.Create("myPage");
    pg.GetWindow().SetFont(SYSTEM_FONT);
    string strFont = pg.GetWindow().FontName;    
    out_str(strFont);
    
    double dSize = pg.GetWindow().FontSize;    
    out_double("",dSize);    
}
