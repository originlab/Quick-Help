Description

Get internal profiling info

Syntax

BOOL GetProfileData( vector<string> & vsFuncNames, vector<uint> & vnNumCalls, vector<double> & vdTimes )

Parameters

vsFuncNames
[output]internal mangled function names
vnNumCalls
[output]number of times corresponding function was called
vdTimes
[output]total amount of time in seconds the corresponding function was called

Return

FALSE if no profiling info available

Examples

EX1

#include <profiler.h>
void Project_GetProfileData_ex1()
{
 vector<string>  vsFuncNames;
 vector<uint>  vnNumCalls;
 vector<double>  vdTimes ;
 Profiler tempObj;
 Worksheet wks = Project.ActiveLayer();
 string strPage;
 if( wks.GetPage().GetName(strPage))
    printf("Active Page is %s; Active Layer is %s\n",strPage ,wks.GetName());
 else 
        return;
 
 if(Project.GetProfileData(vsFuncNames,vnNumCalls,vdTimes))
    printf("Failed to get Profile data!\n");
}
