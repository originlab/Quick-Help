Description

a little utility function to print out byte vector

Syntax

void vector_out( const vector<byte> & vn )

Parameters

vn
[input] the vector to be printed

Return

Examples

EX1

void BitsHex_vector_out_ex1()
{
        string strHex = "E38C";
        vector<byte> vb;
        BitsHex bh;
        if ( bh.HexStrToBits(strHex, vb) )
        {
                printf("Hex string %s is convert to vector as follows:\n", strHex);
                bh.vector_out(vb); //output a string containing '0's and '1's
        }
        else
                printf("Fail to convert Hex string %s to byte vector.\n", strHex);
}
