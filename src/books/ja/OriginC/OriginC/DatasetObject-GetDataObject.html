Description

to access (read/write) the dataset in DatasetObject.

Syntax

vectorbase& GetDataObject()

Parameters

Return

A reference to the vectorbase class object for the associated dataset in the DatasetObject

Examples

EX1

void DatasetObject_GetDataObject_Ex1()
{
    Worksheet wks = Project.ActiveLayer();
    if( !wks )
        return;

    Column col = wks.Columns(0);
    DatasetObject obj(col);
    
    if( obj )
    {
        // Note here used "vector<double>" as the data type of variable vData. The data type of vector
        // must keep consistent with the data type of the DatasetObject.
        vector<double>& vData = obj.GetDataObject();

        out_int("Size = ", vData.GetSize());
    }
}
