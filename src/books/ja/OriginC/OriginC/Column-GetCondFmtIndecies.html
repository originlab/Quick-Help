Description

Get indexes of conditional formatted rows

Syntax

void GetCondFmtIndecies( vector<int> & vnIndex )

Parameters

vnIndex
[output]row indexes, 0 offset

Return

Examples

EX1

void GetCondFmtIndecies_ex()
{
        int nCol = 0;
        
        Worksheet wks;
        wks.Create("origin");
        vector vv;
        vv.Data(1, 5, 1);
        Dataset dsA(wks, nCol);
        dsA = vv;
        
        //format cells > 3 in active sheet
        LT_execute("wcolor trrule.cond1:=2 trrule.val11:=3");
        
        Column colFmt(wks, nCol);
        vector<int> vn;
        colFmt.GetCondFmtIndecies(vn);
        int nRet = vn.GetSize();
        if(nRet < 1)
        {
                out_str("no result");
                return;
        }
        
        printf("rows:");
        for(int ii = 0; ii < nRet; ii++)
        {
                printf(" %d", vn[ii]);
        }
}
