Description

Save a Project page to an enhanced meta file (EMF).

Syntax

BOOL SaveEMF( LPCSTR lpcszEMFFile )

Parameters

lpcszEMFFile
[input]Full path of emf to save to

Return

Returns TRUE on success and FALSE on failure.

Examples

EX1

int PageBase_SaveEMF_ex1()
{
    // Make sure at least one graph page exists.
    GraphPage pg = Project.GraphPages(0);
    if( !pg )
    {
        out_str("Invalid page!\n");
        return 0;
    }
    // For this to succeed the folder c:\ must exist
    BOOL bOK = pg.SaveEMF("c:\\Origin Graph.EMF");
    out_int("OK = ", bOK);
    return 0;
}
