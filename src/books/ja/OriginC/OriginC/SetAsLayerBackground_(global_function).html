Description

Set inserted image as background.

Syntax

BOOL SetAsLayerBackground( LPCSTR lpcszGrRange, DWORD dwCntrl )

Parameters

lpcszGrRange
[input] The inserted image object in graph layer
dwCntrl
[input] set SLBG_APPLY_XYSCALE to rescale Layer using Image Coordinates

Return

true if success, else false.

Examples

EX1 

//assum graph1 has image object "CVIMG"
void SetAsBackground_ex()
{
	GraphLayer gl("Graph1");
	if(!gl)
		return;
	
	GraphObject go = gl.GraphObjects("CVIMG");
	if(!go)
		return;
	
	string strRange;
	go.GetRangeString(strRange);
	out_str(strRange);

	BOOL bRet = SetAsLayerBackground(strRange, 0);
}
