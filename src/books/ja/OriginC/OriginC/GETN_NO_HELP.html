Name

GETN_NO_HELP

Declaration

#define GETN_NO_HELP                _tmpNode.SetAttribute(STR_HELPID_ATTRIB, 0);

Remark

Set 0 to Help ID. No question button on the upper right of the dialog.

Parameters

Return

Examples

EX1

#include <GetNBox.h>
void GetNBox_ex1()
{
    double x0, x1;
    GETN_BOX(trGetN)
    GETN_NO_HELP
    GETN_NUM(xFrom, "X From", 1.3)
    GETN_NUM(xStep, "X Step", -0.5)
    if(GetNBox(trGetN, "Row# as X", "Please specify initial X value and increment"))
    {
        x0 = trGetN.xFrom.dVal;
        x1 = trGetN.xStep.dVal;
        printf("X from %f with increment %f\n", x0, x1);
    }
}
