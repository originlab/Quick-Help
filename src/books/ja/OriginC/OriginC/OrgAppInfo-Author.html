void InstalledAppsExample()
{
    foreach(OrgAppInfo app in OrgApps.InstalledApps)
    {
        if (app.ID <= 0)
            continue;
        printf("[ID: %d]\r\n", app.ID);
        printf("  Name: %s\r\n", app.Name);
        printf("  Description: %s\r\n", app.Description);
        printf("  Author: %s\r\n", app.Author);
        printf("  UpdatedDate: %s\r\n", app.UpdatedDate);
        printf("  Version: %lf\r\n", app.Version);
        printf("  MinOriginVersion: %s\r\n", app.MinOriginVersion);
        printf("  URL: %s\r\n", app.URL);
        printf("  IconDataURL: %s\r\n", app.IconDataURL);
        printf("  HasUpdate: %d\r\n", app.HasUpdate);
        printf("  IconPath: %s\r\n", app.GetIconPath(FALSE));
    }
}
