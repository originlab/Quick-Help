EX1

int Window_Window_ex1()
{
    Window wnd;
    Page pb;
    pb = Project.Pages(); // Get the project's active page
    wnd = pb.GetWindow();
    
    // hide window
    if(wnd)
        wnd.Visible  = false;
    
    // show window
    if(wnd)
        wnd.Visible  = true;
    
    return 0;
}
