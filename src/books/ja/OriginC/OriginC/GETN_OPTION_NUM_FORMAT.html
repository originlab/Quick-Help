Name

GETN_OPTION_NUM_FORMAT

Declaration

#define GETN_OPTION_NUM_FORMAT(_STRFMT)   _tmpSubNode.SetAttribute(STR_ATTRIB_NUMFMT, _STRFMT);

Remark

Custom the showing numeric type in the dialog box.It affect the whole dialog box now.

Parameters

_STRFMT
LabTalk formatting string, default (NULL) will use "*".

Return

Examples

EX1

#include <GetNbox.h>
void GETN_OPTION_NUM_FORMAT_ex1()
{
    GETN_TREE( treeTest )
    double xx = pi;
    GETN_NUM(testVal, "value in 2 decimal", xx)
    GETN_OPTION_NUM_FORMAT( ".2" )

    // gets current time
    SYSTEMTIME st;
    GetSystemTime(&st);
    double dDate;
    SystemTimeToJulianDate(&dDate, &st);

    GETN_NUM(testDate, "value in date format", dDate)
    GETN_OPTION_NUM_FORMAT( "D9" ) // 0 offset date format drop down, see LT manual
    
    if( GetNBox( treeTest ))
        out_tree( treeTest );
}
