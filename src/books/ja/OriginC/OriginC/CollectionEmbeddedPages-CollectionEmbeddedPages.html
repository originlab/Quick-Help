Description

The default constructor.


The constructor from Worksheet.

Syntax

CollectionEmbeddedPages( )


CollectionEmbeddedPages( Worksheet & wks, DWORD dwExcludeBits = 0 )

Parameters

wks
[input] The Worksheet which contains the Embeddings
dwExcludeBits
[input] Used to filter pages. Embedded pages that have these bits are excluded from the collection.
Could be one or more of the following bits
EMBEDGRAPH_KEEP_ASPECT_RATIO,
EMBEDGRAPH_IN_LABELS,
EMBEDGRAPH_DO_UNDO,
EMBEDGRAPH_DISPLAY_MASK,
EMBEDGRAPH_HIDE_AXES,
EMBEDGRAPH_HIDE_LEGENDS,
EMBEDGRAPH_HIDE_SPECTRUMS,
EMBEDGRAPH_HIDE_TEXT_OBJS,
EMBEDGRAPH_SPARKLINE_LARGE,
EMBEDGRAPH_SPARKLINE_CLIPBOARD,
EMBEDGRAPH_SPARKLINE_MASK,
EMBEDGRAPH_CACHE_IN_PAGE,
EMBEDGRAPH_NO_RESIZE_CELL,
EMBEDGRAPH_MOVE_IF_ALREADY_EMBEDDED,
EMBEDGRAPH_NO_PASTE_FROM_CLIPBOARD

Return

Examples

EX1

void CollectionEmbeddedPages_CollectionEmbeddedPages_Ex1()
{
    Worksheet wks = Project.ActiveLayer();
    if(!wks)
        return;
    
    CollectionEmbeddedPages embeddedPages;
    embeddedPages.Attach(wks);
    int nCount = embeddedPages.Count();
    printf("%d pages embdded in a worksheet", nCount);
}


EX2

void CollectionEmbeddedPages_CollectionEmbeddedPages_Ex2()
{
    Worksheet wks = Project.ActiveLayer();
    if(!wks)
        return;
    
    CollectionEmbeddedPages embeddedPages(wks);
    int nCount = embeddedPages.Count();
    printf("%d pages embdded in a worksheet", nCount);
}
