Description

Get the image export settings from a page.

Syntax

bool tree_get_page_image_export_settings( TreeNode & trSettings, Page & pg, LPCSTR lpcszFormat )

Parameters

trSettings
[input]tree node to receive the settings
pg
[input]page to get the settings from
lpcszFormat
[input]pointer to the image format

Return

true for success, false for error

Examples

EX1

//Please run example code of tree_set_page_image_export_settings before this code to check the result
void tree_get_page_image_export_settings_ex1()
{
        Tree trSettings;
        Page pg = Project.Pages(); //active page
        if ( !pg )
                return;
        
        bool bRet = tree_get_page_image_export_settings(trSettings, pg, "bmp");
        if ( bRet )
        {
                out_str("Get page image export settings successfully.");
                out_tree(trSettings);
        }
        else
                out_str("Fail to get page image export settings.");
        
        return;
}
