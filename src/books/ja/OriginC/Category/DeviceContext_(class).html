int DeviceContext_ex1()
{
    Page pb;
    pb = Project.Pages(); // Get the project's active page
    if( pb.IsValid() )
    {
        Window    myWindow = pb.GetWindow();
        DeviceContext myDC = myWindow.GetDC();
        int nPixels = myDC.GetDeviceCaps(LOGPIXELSX);
    }
    return 1;
}
